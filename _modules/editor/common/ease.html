	c3 = c1 + 1
	return c3 * x * x * x - c1 * x * x


[docs]def easeOutBack(x):

[docs]def easeOutBounce(x):
	n1 = 7.5625
	d1 = 2.75
	if x < 1 / d1:
		return n1 * x * x
	elif x < 2 / d1:
		x = x - 1.5
		return n1 * (x / d1) * x + 0.75
	elif x < 2.5 / d1:
		x = x - 2.25
		return n1 * (x / d1) * x + 0.9375
	else:
		x = x - 2.625
		return n1 * (x / d1) * x + 0.984375


[docs]def easeInOutBounce(x):
	if x < 0.5: return (1 - easeOutBounce(1 - 2 * x)) / 2
	else: return (1 + easeOutBounce(2 * x - 1)) / 2
